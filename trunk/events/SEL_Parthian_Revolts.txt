# Parthian revolts

country_event = {
	id = seleucid.1
	title = "seleucid.EVTNAME1"
	desc = "seleucid.EVTDESC1"
	picture = ANGRY_MOB_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = SEL
		any_owned_province = {is_core = PRT}
		NOT = { exists = PRT }
		is_year = 0500
	}
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "seleucid.EVTOPTA1"
		every_owned_province = {
				limit = { is_core = PRT }

				spawn_rebels = {
					type = nationalist_rebels
					size = 2
					leader = "Arsaces I of Parthia"
				}
				
		}
	}
